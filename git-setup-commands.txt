# Git Setup Commands
# Run these commands in your terminal after installing Git

# 1. Configure your Git identity
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# 2. Navigate to your project directory
cd "C:\Users\Skyla barbosa\OneDrive\Desktop\Odoo-x-Amalthea--IIT-GN-Hackathon-2025-main"

# 3. Initialize Git repository
git init

# 4. Add all files to staging
git add .

# 5. Create initial commit
git commit -m "Initial commit: Expense Management Application with modern CSS styling"

# 6. Create a new repository on GitHub (go to github.com and create a new repository)

# 7. Add remote origin (replace YOUR_USERNAME and YOUR_REPO_NAME)
git remote add origin https://github.com/YOUR_USERNAME/YOUR_REPO_NAME.git

# 8. Push to GitHub
git branch -M main
git push -u origin main
